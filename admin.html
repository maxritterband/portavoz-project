Setup & Configuration
Connect this admin panel to your Google Sheet.
⚠️ One-time setup required

To save content from this admin panel to your Google Sheet, you need to create a Google Apps Script — a free mini-server hosted by Google. This only needs to be done once. Follow the steps below.

Step 1 · Create the Google Apps Script
  1. Go to script.google.com and click New Project
  2. Delete any existing code in the editor
  3. Copy and paste the script code shown below into the editor
  4. Replace YOUR_SPREADSHEET_ID with your actual Google Sheet ID (the long string in the sheet's URL)
  5. Click Save (give the project any name, e.g. "Portavoz Admin")
  6. Click Deploy → New deployment
  7. Choose type: Web App
  8. Set "Execute as" to Me and "Who has access" to Anyone
  9. Click Deploy — you'll be asked to authorize. Click through.
  10. Copy the Web App URL that appears — it will look like https://script.google.com/macros/s/ABC.../exec
const SPREADSHEET_ID = 'YOUR_SPREADSHEET_ID';
const TRANSLATIONS_SHEET = 'Sheet1'; // your main sheet tab name
const EVENTS_SHEET = 'events';        // your events sheet tab name

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const { type, row } = data;
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

    if (type === 'translation') {
      const sheet = ss.getSheetByName(TRANSLATIONS_SHEET);
      const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      const newRow = headers.map(h => row[h] !== undefined ? row[h] : '');
      sheet.appendRow(newRow);
    } else if (type === 'event') {
      const sheet = ss.getSheetByName(EVENTS_SHEET);
      const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      const newRow = headers.map(h => row[h] !== undefined ? row[h] : '');
      sheet.appendRow(newRow);
    } else if (type === 'student') {
      const sheet = ss.getSheetByName(TRANSLATIONS_SHEET);
      sheet.appendRow([
        '', // id
        '', // title (empty - student-only row)
        '', // author
        '', // semester
        row.studentTranslator || '',
        row.studentClass || '',
        '', // status
        '', // type
        '', // originalTitle
        '', // excerpt
        '', // buyLink
        '', // publisher
        '', // datePublished
        '', // authorBio
        '', // translatorNote
        row.studentPhoto || '',
        row.studentDesc || ''
      ]);
    }

    return ContentService
      .createTextOutput(JSON.stringify({ success: true }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet() {
  return ContentService
    .createTextOutput(JSON.stringify({ status: 'Portavoz Admin API is running' }))
    .setMimeType(ContentService.MimeType.JSON);
}
